class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
    map<int, int> indexMap;

    for (int i = 0; i < nums.size(); i++) {
        int complement = target - nums[i];

        if (indexMap.find(complement) != indexMap.end()) {
            return { i, indexMap[complement] };
        }

        indexMap[nums[i]] = i;
    }
        
    return {};
    
    }

};
