// PROBLEM OF THE DAY: Product of Array Except Self

// PROBLEM LINK: https://leetcode.com/problems/product-of-array-except-self/

 // APPROACH:
        // 1. We are given an array nums and need to compute an array where result[i] is the product of all elements 
        //except nums[i].
        // 2. We use the output array to store prefix products.
        // 3. In the first pass, we store the product of all elements to the left of each index.
        // 4. In the second pass (right to left), we multiply the suffix product with the stored prefix.
        // 5. This avoids using division and extra space.

// SPACE COMPLEXITY: O(1) As output array does not count as extra space.

// TIME COMPLEXITY: O(N) As we are iterating over the array

class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int n =  nums.size();
        vector<int> result(n, 0);
        
        int prefix = 1;
        for(int i = 0; i < n; i++) {
            result[i] = prefix;
            prefix *= nums[i];
        }

        int suffix = 1;
        for(int i = n - 1; i >= 0; i--) {
            result[i] *= suffix;
            suffix *= nums[i];
        }

        return result;
    }
};
